#' Extract a substring block from a hex string
#'
#' @param s Character scalar. Full hex string.
#' @param pos Integer. 1-based start position in `s`.
#' @param n Integer. Number of characters to extract.
#' @return Character scalar of length `n` (or shorter if `s` ends early).
hex_block <- function(s, pos, n) substr(s, pos, pos + n - 1L)


#' Convert hexadecimal string to an unsigned number (robust)
#'
#' Robustly converts a hexadecimal string to a numeric value.
#' This avoids potential overflow issues of `strtoi()` for unsigned 32-bit values
#' (e.g., Unix time encoded as 8 hex chars).
#'
#' @param h Character scalar. Hex string (e.g., "1A18", "FFFFFFFF").
#' @return Numeric scalar (or `NA_real_` if parsing fails).
#' @examples
#' hex_to_uint("1a18")     # 6680
#' hex_to_uint("FFFFFFFF") # 4294967295
hex_to_uint <- function(h) {
  if (is.na(h) || h == "" || nchar(h) == 0L) return(NA_real_)
  h <- toupper(h)

  chars <- strsplit(h, "", fixed = TRUE)[[1]]
  # Map '0'..'9','A'..'F' -> 0..15
  nibble <- match(chars, c(as.character(0:9), LETTERS[1:6])) - 1L
  if (any(is.na(nibble))) return(NA_real_)

  # Accumulate in base-16 (numeric is exact up to 2^53)
  v <- 0
  for (d in nibble) v <- v * 16 + d
  v
}


#' Convert hexadecimal string to signed integer (two's complement)
#'
#' Interprets an unsigned hex value as a signed two's-complement integer
#' of given bit width.
#'
#' @param h Character scalar. Hex string.
#' @param bits Integer. Bit width (e.g., 16, 24).
#' @return Integer scalar (or `NA_integer_` if parsing fails).
#' @examples
#' hex_to_int("FF", 8)     # -1
#' hex_to_int("8000", 16)  # -32768
hex_to_int <- function(h, bits) {
  v <- hex_to_uint(h)
  if (is.na(v)) return(NA_integer_)

  sign_bit <- 2^(bits - 1L)
  if (v >= sign_bit) v <- v - 2^bits

  as.integer(v)
}


#' Get decoding specification for a field name
#'
#' Returns the number of hex characters (`n_hex`) and a decoding function
#' to convert the corresponding block into a value.
#'
#' Rules implemented:
#' - `time_unix`: 8 hex chars, unsigned
#' - `adc_temp`: 6 hex chars, signed 24-bit
#' - `battery_mv`: 4 hex chars, unsigned (uint16)
#' - `*.gnd`: 4 hex chars, signed 16-bit
#' - `*.ntc1`, `*.ntc2`, `*.pressure`: 6 hex chars, signed 24-bit
#'
#' @param name Character scalar. Field name from the schema.
#' @return List with elements `n_hex` (integer) and `decode` (function).
field_spec <- function(name) {
  if (name == "time_unix") {
    return(list(n_hex = 8L, decode = hex_to_uint))
  }
  if (name == "adc_temp") {
    return(list(n_hex = 6L, decode = function(h) hex_to_int(h, bits = 24L)))
  }
  if (name == "battery_mv") {
    return(list(n_hex = 4L, decode = hex_to_uint))
  }

  if (grepl("\\.gnd$", name)) {
    return(list(n_hex = 4L, decode = function(h) hex_to_int(h, bits = 16L)))
  }
  if (grepl("\\.(ntc1|ntc2|pressure|testSB)$", name)) {
    return(list(n_hex = 6L, decode = function(h) hex_to_int(h, bits = 24L)))
  }

  stop("No decoding rule for field: ", name)
}


#' Decode a hex-encoded record according to a schema
#'
#' Decodes a single hex string into named values according to a schema that
#' specifies the order and names of fields. Fields are separated by whitespace.
#'
#' @param schema Character scalar. Whitespace-separated field names.
#'   Example: `"time_unix adc_temp n1.ntc1 n1.ntc2 n1.gnd n1.pressure battery_mv"`.
#' @param hex_string Character scalar. The full hex payload to decode.
#' @return Named list with one element per schema field.
#' @examples
#' schema <- "time_unix adc_temp n1.ntc1 n1.ntc2 n1.gnd n1.pressure battery_mv"
#' # decoded <- decode_hex_line(schema, "6971e7d2...1a18")
decode_hex_line <- function(schema, hex_string) {
  fields <- strsplit(trimws(schema), "\\s+")[[1]]
  out <- vector("list", length(fields))
  names(out) <- fields

  pos <- 1L
  for (i in seq_along(fields)) {
    sp <- field_spec(fields[i])
    block <- hex_block(hex_string, pos, sp$n_hex)

    out[[i]] <- if (nchar(block) == sp$n_hex) sp$decode(block) else NA
    pos <- pos + sp$n_hex
  }

  # Warn if extra characters remain (usually indicates schema mismatch)
  if (pos <= nchar(hex_string)) {
    leftover <- substr(hex_string, pos, nchar(hex_string))
    if (nchar(trimws(leftover)) > 0L) {
      warning("Unused trailing hex characters: ", leftover)
    }
  }

  out
}

Code for plausibility check of the data (are the outputs reasonable); If e.g. a byte is shifted, usually this will fail and is thus a useful check of the integrity

#' Check plausibility of decoded sensor values
#'
#' Performs a simple plausibility check on decoded sensor data.
#' Ground reference values (`*.gnd`) must lie between -1000 and -500,
#' and pressure values (`*.pressure`) between 620000 and 740000.
#'
#' If all values fall within the expected ranges, a message "Plausibility check: OK"
#' is printed. Otherwise a warning is issued.
#'
#' @param decoded Named list returned by `decode_hex_line()`.
#'
#' @return Invisibly returns `NULL`. The function is intended for quick validation.
#'
#' @examples
#' decoded <- decode_hex_line(schema, hex_string)
#' check_plausibility(decoded)
check_plausibility <- function(decoded) {
  gnd  <- unlist(decoded[grep("\\.gnd$", names(decoded))])
  pres <- unlist(decoded[grep("\\.pressure$", names(decoded))])

  if (all(gnd >= -1000 & gnd <= -500, na.rm = TRUE) &&
      all(pres >= 620000 & pres <= 740000, na.rm = TRUE)) {
    message("Plausibility check: OK")
  } else {
    warning("Plausibility check failed")
  }
}

Conversion codes for the fields

#' Convert ADC counts to temperature
#'
#' Linear conversion from ADC counts to temperature (°C).
#'
#' @param count Numeric vector of ADC counts.
#' @return Numeric vector of temperatures in °C.
adc2temp <- function(count) {
  6.101e-5 * count - 256.6
}

#' Convert NTC ADC counts to uncalibrated temperature
#'
#' Converts NTC counts to temperature (°C) using the sensor/resistor model
#' implemented in the logger firmware.
#'
#' @param counts Numeric vector of NTC counts.
#' @return Numeric vector of uncalibrated temperatures in °C.
NTCcounts2temp <- function(counts) {
  H <- (-counts * 1000000) / (counts - 33554432)
  resistance <- (H * 499000) / (499000 - H)
  temperatures <- 1 / ((1 / 3380) * log(resistance / 10000) + 1 / 298.15)
  temperatures <- temperatures - 273.15
  temperatures
}

#' Convert pressure counts to hPa
#'
#' Converts raw pressure values to hectopascal (hPa) by dividing by 10000.
#'
#' @param pressure Numeric vector of raw pressure values.
#' @return Numeric vector of pressure in hPa.
pressure2hPa <- function(pressure) {
  pressure / 1000
}

#' Convert Unix time to POSIXct (UTC)
#'
#' Converts Unix epoch seconds to a POSIXct timestamp in UTC.
#'
#' @param time_unix Numeric vector of Unix time (seconds since 1970-01-01).
#' @return POSIXct vector in UTC.
unix2posix_utc <- function(time_unix) {
  as.POSIXct(time_unix, origin = "1970-01-01", tz = "UTC")
}
